## 二分图检测

```c++
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

// 图的邻接表表示
vector<vector<int>> adj;

// 标记节点的颜色,0表示未访问,1表示红色,2表示蓝色
vector<int> color;

// BFS函数用于检测是否为二分图
bool bfs(int start) {
    queue<int> q;
    q.push(start);
    color[start] = 1; // 起点颜色为红色

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        // 遍历u的邻居节点
        for (int v : adj[u]) {
            // 如果邻居节点未被访问过,则给它涂色
            if (color[v] == 0) {
                color[v] = 3 - color[u]; // 邻居节点颜色与u不同
                q.push(v);
            }
            // 如果邻居节点与u颜色相同,则不是二分图
            else if (color[v] == color[u]) {
                return false;
            }
        }
    }

    return true;
}

bool isBipartite(int n) {
    color.assign(n, 0); // 初始化颜色向量

    // 对每个未访问的节点执行BFS
    for (int i = 0; i < n; i++) {
        if (color[i] == 0) {
            if (!bfs(i)) {
                return false;
            }
        }
    }

    return true;
}

int main() {
    int n, m;
    cout << "Enter the number of nodes and edges: ";
    cin >> n >> m;

    adj.resize(n); // 初始化邻接表

    cout << "Enter the edges: " << endl;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u); // 无向图需要双向添加边
    }

    if (isBipartite(n)) {
        cout << "The graph is bipartite." << endl;
    } else {
        cout << "The graph is not bipartite." << endl;
    }

    return 0;
}
```
